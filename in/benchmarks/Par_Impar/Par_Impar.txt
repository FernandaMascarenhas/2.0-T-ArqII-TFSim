LD R1, 0(R0)      // Carrega o valor do número na memória para o registrador R1
DADDI R2, R1, -1  // Subtrai 1 do número
ANDI R2, R2, 1    // Aplica uma operação AND lógica com 1 para manter apenas o bit menos significativo

BNE R2, R0, 2 // Se o resultado for diferente de 0, salta para o rótulo IMPAR
J 2             // Caso contrário, salta para o rótulo FIM
    
//IMPAR:
DADDI R3, R0, 1   // Carrega o valor 1 para o registrador R3
    
//FIM:
SD R3, 0(R0)      // Armazena o resultado no endereço de memória 0
